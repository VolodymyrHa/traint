pipeline {
  agent none
  stages {
    stage('Nginx') {
      agent {
        docker { image 'nginx' }
      }
      steps {
        sh 'ls'
      }
    }
    stage('Tomcat') {
      agent {
        docker { image 'tomcat:9.0' }
      }
      steps {
        
        sh 'docker run -it --rm -p 8888:8181 tomcat:9.0'
        sh 'catalina.sh'
        sh 'run'        
      }
    }
    
    }
  }
